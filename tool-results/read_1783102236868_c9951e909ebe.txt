   227→# ===========================================================================
   228→# HTTP helper con anti-blocking
   229→# ===========================================================================
   230→def _fetch_url(url: str, timeout: int = 15) -> Optional[str]:
   231→    """Fetch URL con random UA, retry y backoff."""
   232→    cache_key = hashlib.sha256(url.encode()).hexdigest()[:16]
   233→    now = time.time()
   234→
   235→    # Check cache
   236→    if cache_key in _cache:
   237→        ts, data = _cache[cache_key]
   238→        if now - ts < CACHE_TTL:
   239→            return data
   240→
   241→    for attempt in range(MAX_RETRIES):
   242→        try:
   243→            req = urllib.request.Request(url)
   244→            req.add_header("User-Agent", random.choice(USER_AGENTS))
   245→            req.add_header("Accept", "text/html,application/xhtml+xml,application/json,*/*")
   246→            req.add_header("Accept-Language", "es-AR,es;q=0.9,en;q=0.8")
   247→
   248→            with urllib.request.urlopen(req, timeout=timeout) as resp:
   249→                content = resp.read().decode("utf-8", errors="replace")
   250→                _cache[cache_key] = (now, content)
   251→                return content
   252→        except Exception as e:
   253→            if attempt < MAX_RETRIES - 1:
   254→                wait = (attempt + 1) * 2 + random.uniform(0, 1)
   255→                time.sleep(wait)
   256→            else:
   257→                return None
   258→    return None

   262→# Provider 1: DuckDuckGo HTML

   264→class _DDGResultParser(HTMLParser):
   265→    """Parser minimalista para resultados de DuckDuckGo HTML."""
   266→
   267→    def __init__(self):
   268→        super().__init__()
   269→        self.results: List[Dict[str, str]] = []
   270→        self._current: Dict[str, str] = {}
   271→        self._in_result = False
   272→        self._in_title = False
   273→        self._in_snippet = False
   274→        self._capture_tag = None
   275→
   276→    def handle_starttag(self, tag, attrs):
   277→        attrs_dict = dict(attrs)
   278→        # DuckDuckGo lite usa <a class="result-link">
   279→        # DuckDuckGo HTML usa <div class="result">
   280→        cls = attrs_dict.get("class", "")
   281→
   282→        if "result-link" in cls or ("result__a" in cls):
   283→            self._in_title = True
   284→            self._current = {"title": "", "url": "", "snippet": ""}
   285→            href = attrs_dict.get("href", "")
   286→            # DDG a veces wrappea URLs
   287→            if href.startswith("//duckduckgo.com/l/?uddg="):
   288→                href = urllib.parse.unquote(href.split("uddg=")[1].split("&")[0])
   289→            self._current["url"] = href
   290→
   291→        if "result-snippet" in cls or "result__snippet" in cls:
   292→            self._in_snippet = True
   293→            if not self._current:
   294→                self._current = {"title": "", "url": "", "snippet": ""}
   295→
   296→    def handle_endtag(self, tag):
   297→        if self._in_title and tag == "a":
   298→            self._in_title = False
   299→        if self._in_snippet and tag in ("a", "td", "div"):
   300→            self._in_snippet = False
   301→            if self._current and self._current.get("url"):
   302→                self.results.append(self._current)
   303→                self._current = {}
   304→
   305→    def handle_data(self, data):
   306→        if self._in_title:
   307→            self._current["title"] += data.strip()
   308→        if self._in_snippet:
   309→            self._current["snippet"] += data.strip() + " "

   312→def search_duckduckgo(query: str, num: int = 10) -> List[Dict[str, Any]]:
   313→    """Busca en DuckDuckGo (HTML lite version)."""
   314→    # Usar DuckDuckGo Lite (html only, más fácil de parsear)
   315→    encoded = urllib.parse.quote(query)
   316→    url = f"https://lite.duckduckgo.com/lite/?q={encoded}&kl=ar-es"
   317→
   318→    html = _fetch_url(url)
   319→    if not html:
   320→        return []
   321→
   322→    parser = _DDGResultParser()
   323→    try:
   324→        parser.feed(html)
   325→    except Exception:
   326→        pass
   327→
   328→    results = []
   329→    for r in parser.results[:num]:
   330→        if r.get("url") and r.get("title"):
   331→            results.append({
   332→                "title": r["title"][:200],
   333→                "url": r["url"],
   334→                "snippet": r.get("snippet", "")[:300],
   335→                "source": "duckduckgo",
   336→                "date": "",
   337→            })
   338→
   339→    return results

   345→def search_reddit(query: str, num: int = 10) -> List[Dict[str, Any]]:
   346→    """Busca en Reddit via JSON público (sin auth)."""
   347→    encoded = urllib.parse.quote(query)
   348→    # Reddit search JSON endpoint (público)
   349→    url = f"https://www.reddit.com/search.json?q={encoded}&sort=new&limit={num}&type=link"
   350→
   351→    # Reddit requiere un UA custom (no genérico)
   352→    try:
   353→        req = urllib.request.Request(url)
   354→        req.add_header("User-Agent", "RadarLeadsBot/1.0 (lead intelligence research)")
   355→        with urllib.request.urlopen(req, timeout=15) as resp:
   356→            data = json.loads(resp.read().decode("utf-8", errors="replace"))
   357→    except Exception:
   358→        return []
   359→
   360→    results = []
   361→    if isinstance(data, dict) and "data" in data:
   362→        for child in data["data"].get("children", []):
   363→            post = child.get("data", {})
   364→            if not post:
   365→                continue
   366→
   367→            # Construir URL completa
   368→            permalink = post.get("permalink", "")
   369→            full_url = f"https://www.reddit.com{permalink}" if permalink else ""
   370→
   371→            # Snippet: selftext si existe, sino title
   372→            selftext = post.get("selftext", "")
   373→            snippet = selftext[:300] if selftext else post.get("title", "")[:300]
   374→
   375→            # Fecha
   376→            created = post.get("created_utc", 0)
   377→            date = ""
   378→            if created:
   379→                from datetime import datetime, timezone
   380→                date = datetime.fromtimestamp(created, tz=timezone.utc).isoformat()
   381→
   382→            results.append({
   383→                "title": post.get("title", "")[:200],
   384→                "url": full_url,
   385→                "snippet": snippet,
   386→                "source": "reddit",
   387→                "date": date,
   388→                "username": post.get("author", ""),
   389→            })
   390→
   391→    return results[:num]

   414→            req = urllib.request.Request(feed_url)

   416→            with urllib.request.urlopen(req, timeout=10) as resp:
   417→                data = json.loads(resp.read().decode("utf-8", errors="replace"))
   418→
   419→            if isinstance(data, dict) and "data" in data:
   420→                for child in data["data"].get("children", []):
   421→                    post = child.get("data", {})
   422→                    if not post:
   423→                        continue
   424→
   425→                    text = f"{post.get('title', '')} {post.get('selftext', '')}".lower()
   426→
   427→                    # Filtro por keywords de la query
   428→                    if query_keywords and not any(kw in text for kw in query_keywords):
   429→                        continue
   430→
   431→                    permalink = post.get("permalink", "")
   432→                    full_url = f"https://www.reddit.com{permalink}" if permalink else ""
   433→
   434→                    created = post.get("created_utc", 0)
   435→                    date = ""
   436→                    if created:
   437→                        from datetime import datetime, timezone

   442→                        "url": full_url,

   444→                        "source": "reddit_rss",
   445→                        "date": date,
   446→                        "username": post.get("author", ""),

   460→# ===========================================================================
   461→# Search Manager — unifica todos los providers
   462→# ===========================================================================
   463→def search(query: str, num: int = 10) -> List[Dict[str, Any]]:
   464→    """
   465→    Busca usando múltiples providers en orden de fallback.
   466→
   467→    Orden:
   468→      1. DuckDuckGo (cobertura amplia)
   469→      2. Reddit (foros, alta calidad de leads)
   470→      3. RSS (backup)
   471→
   472→    Devuelve resultados normalizados con campos:
   473→      title, url, snippet, source, date, username?
   474→    """
   475→    all_results = []
   476→
   477→    # Provider 1: DuckDuckGo
   478→    try:
   479→        ddg = search_duckduckgo(query, num=num)
   480→        all_results.extend(ddg)
   481→        time.sleep(RATE_LIMIT_SECONDS)
   482→    except Exception:
   483→        pass
   484→
   485→    # Provider 2: Reddit (sólo si la query no es muy larga)
   486→    if len(query) < 200:
   487→        try:
   488→            reddit = search_reddit(query, num=num)
   489→            all_results.extend(reddit)
   490→            time.sleep(RATE_LIMIT_SECONDS)
   491→        except Exception:
   492→            pass
   493→
   494→    # Provider 3: RSS (sólo si los anteriores no dieron suficiente)
   495→    if len(all_results) < num // 2:
   496→        try:
   497→            rss = search_rss(query, num=num)
   498→            all_results.extend(rss)
   499→        except Exception:
   500→            pass
   501→
   502→    # Dedup por URL
   503→    seen_urls = set()
   504→    unique = []
   505→    for r in all_results:
   506→        url = r.get("url", "")
   507→        if url and url not in seen_urls:
   508→            seen_urls.add(url)
   509→            unique.append(r)
   510→
   511→    return unique[:num]

   787→def score_lead(text: str, date: str, has_contact: bool) -> Dict[str, Any]:
   788→    """Scoring con 5 signals independientes, cada uno 0-100."""
   789→    text_lower = text.lower()
   790→
   791→    # Signal 1: intent (lenguaje de intención explícita)
   792→    intent = 20
   793→    if any(kw in text_lower for kw in ["no puedo", "necesito", "quiero", "ayuda", "alguien sabe"]):
   794→        intent = 80
   795→    elif any(kw in text_lower for kw in ["consulta", "duda", "pregunta"]):
   796→        intent = 60
   797→
   798→    # Signal 2: urgency
   799→    urgency = 20
   800→    if any(kw in text_lower for kw in ["urgente", "hoy", "ya", "inmediato"]):
   801→        urgency = 90
   802→    elif any(kw in text_lower for kw in ["mañana", "vencimiento", "antes de"]):
   803→        urgency = 65
   804→
   805→    # Signal 3: financial_pressure
   806→    financial = 20
   807→    if any(kw in text_lower for kw in ["debo", "deuda", "multa", "$", "pesos", "plata"]):
   808→        financial = 75
   809→    if any(kw in text_lower for kw in ["bloqueada", "embargo", "inhibición"]):
   810→        financial = 85
   811→
   812→    # Signal 4: legal_risk
   813→    legal = 20
   814→    if any(kw in text_lower for kw in ["juzgado", "notificación", "citación", "inhibición"]):
   815→        legal = 85
   816→    elif any(kw in text_lower for kw in ["rechazaron", "bloqueada", "prohibido"]):
   817→        legal = 65
   818→
   819→    # Signal 5: recency
   820→    recency = 20
   821→    dt = _parse_date(date)
   822→    if dt:
   823→        now = datetime.now(timezone.utc)
   824→        if dt.tzinfo is None:
   825→            dt = dt.replace(tzinfo=timezone.utc)
   826→        diff = now - dt
   827→        if diff <= timedelta(days=3):
   828→            recency = 90
   829→        elif diff <= timedelta(days=7):
   830→            recency = 60
   831→
   832→    # Total: weighted sum clamped 0-100
   833→    # v6 weights exactos del spec
   834→    total = (
   835→        intent * 0.35 +
   836→        urgency * 0.20 +
   837→        financial * 0.20 +
   838→        legal * 0.15 +
   839→        recency * 0.10
   840→    )
   841→
   842→    # v6 modifiers
   843→    modifiers = {}
   844→    if has_contact:
   845→        modifiers["public_contact_visible"] = 15
   846→        total += 15
   847→    # TODO: multi_source_confirmation +25 when same entity appears in multiple sources
   848→    modifiers["multi_source_confirmation"] = 0
   849→
   850→    # v6 decay function: score * exp(-days_since_post / 7)
   851→    dt = _parse_date(date)
   852→    if dt:
   853→        now = datetime.now(timezone.utc)
   854→        if dt.tzinfo is None:
   855→            dt = dt.replace(tzinfo=timezone.utc)
   856→        days = (now - dt).total_seconds() / 86400
   857→        import math
   858→        decay = math.exp(-days / 7)
   859→        total = total * decay
   860→        modifiers["decay_factor"] = round(decay, 3)
   861→
   862→    total = max(0, min(100, int(round(total))))
   863→
   864→    # Penalties
   865→    penalties = {"duplication_risk": 0, "noise_probability": 0}
   866→    if not has_contact:
   867→        penalties["noise_probability"] = 30
   868→
   869→    # v7: multipliers
   870→    source_mult = 1.0  # default
   871→    # intent_multiplier from taxonomy
   872→    intent_mult = 1.0
   873→
   874→    return {
   875→        "total": total,
   876→        "base_score": 20,
   877→        "intent_multiplier": round(intent_mult, 2),
   878→        "source_multiplier": round(source_mult, 2),
   879→        "recency_multiplier": round(recency / 100, 2),
   880→        "final_score": total,
   881→        "signals": {
   882→            "intent": intent,
   883→            "urgency": urgency,
   884→            "financial_pressure": financial,
   885→            "legal_risk": legal,
   886→            "recency": recency,
   887→        },
   888→        "penalties": penalties,
   889→    }

   892→def _parse_date(date_str: str) -> Optional[datetime]:
   893→    if not date_str:
   894→        return None
   895→    for fmt in ["%Y-%m-%dT%H:%M:%S%z", "%Y-%m-%dT%H:%M:%SZ", "%Y-%m-%d", "%b %d, %Y"]:
   896→        try:
   897→            return datetime.strptime(date_str[:25], fmt)
   898→        except ValueError:
   899→            continue

  1011→def normalize_entity(result: Dict[str, Any], query: str) -> Optional[Dict[str, Any]]:
  1012→    """Layer 2: normaliza un resultado en un lead + entity."""
  1013→    name = result.get("name", "")
  1014→    snippet = result.get("snippet", "")
  1015→    url = result.get("url", "")
  1016→    date = result.get("date", "")
  1017→    source = result.get("source", "unknown")
  1018→    username = result.get("username", "")
  1019→    combined = f"{name}. {snippet}"
  1020→    combined_lower = combined.lower()
  1021→    host = get_host(url)
  1022→
  1023→    # Content filter
  1024→    if not any(kw in combined_lower for kw in MUST_INCLUDE):
  1025→        return None
  1026→    for reject in REJECT_KW:
  1027→        if reject in combined_lower:
  1028→            return None
  1029→    if any(d in host for d in INSTITUTIONAL_DOMAINS):
  1030→        return None
  1031→
  1032→    # Country filter
  1033→    for country, indicators in FOREIGN_INDICATORS.items():
  1034→        for ind in indicators:
  1035→            if ind in combined_lower:
  1036→                return None
  1037→
  1038→    has_arg = any(s in combined_lower for s in ARG_SIGNALS)
  1039→    if not has_arg:
  1040→        has_arg = True  # assume Argentina if no foreign signal
  1041→
  1042→    # v6: source_type para policy engine
  1043→    if "facebook" in host:
  1044→        source_type = "facebook_group"
  1045→    elif "reddit" in host:
  1046→        source_type = "reddit"
  1047→    elif "mercadolibre" in host:
  1048→        source_type = "marketplace"
  1049→    elif "twitter" in host or "x.com" in host:
  1050→        source_type = "x"
  1051→    else:
  1052→        source_type = "other"
  1053→
  1054→    # Extract contact
  1055→    phone = ""
  1056→    for pattern in PHONE_PATTERNS:
  1057→        m = re.search(pattern, combined)
  1058→        if m:
  1059→            digits = re.sub(r"\D", "", m.group(0))
  1060→            if 10 <= len(digits) <= 15:
  1061→                phone = m.group(0).strip()
  1062→                break
  1063→
  1064→    whatsapp = ""
  1065→    for pattern in WHATSAPP_PATTERNS:
  1066→        m = re.search(pattern, combined, re.IGNORECASE)
  1067→        if m:
  1068→            num = m.group(1) if m.groups() else m.group(0)
  1069→            digits = re.sub(r"\D", "", num)
  1070→            if 8 <= len(digits) <= 15:
  1071→                whatsapp = digits
  1072→                break
  1073→
  1074→    email_match = re.search(r'[\w.+-]+@[\w-]+\.[\w.]+', combined)
  1075→    email = email_match.group(0) if email_match else ""
  1076→
  1077→    contact = {}
  1078→    contactability_score = 0  # 0-100, no boolean
  1079→    if phone:
  1080→        contact["phone"] = phone
  1081→        contact["phone_detected"] = True
  1082→        contactability_score += 40
  1083→    else:
  1084→        contact["phone_detected"] = False
  1085→    if whatsapp:
  1086→        wa_num = whatsapp
  1087→        if not wa_num.startswith("54"):
  1088→            wa_num = "54" + wa_num.lstrip("0")
  1089→        contact["whatsapp"] = f"https://wa.me/{wa_num}"
  1090→        contact["whatsapp_detected"] = True
  1091→        contactability_score += 50
  1092→    else:
  1093→        contact["whatsapp_detected"] = False
  1094→    if email:
  1095→        contact["email"] = email
  1096→        contactability_score += 20
  1097→
  1098→    # v7: contactability_score considera también la fuente
  1099→    source_trust = SOURCE_TRUST.get(source_type, SOURCE_TRUST["unknown"])["trust"]
  1100→    contactability_score = int(contactability_score * source_trust)
  1101→    contact["contactability_score"] = min(100, contactability_score)
  1102→    contact["whatsapp_validated"] = False  # nunca validar contra WA API — sólo probabilidad
  1103→
  1104→    # Create entity (person)
  1105→    entity_name = username or name[:30] or "(anónimo)"
  1106→    entity = create_entity("person", entity_name, combined)
  1107→
  1108→    # Platform
  1109→    platform_map = {"facebook.com": "Facebook", "reddit.com": "Reddit",
  1110→                    "twitter.com": "X", "x.com": "X", "mercadolibre.com.ar": "MercadoLibre"}
  1111→    platform = platform_map.get(host, host.title() if host else "Unknown")
  1112→
  1113→    # Province
  1114→    provincia = ""
  1115→    for prov_key in ["buenos aires", "caba", "santa fe", "rosario", "córdoba", "cordoba",
  1116→                      "mendoza", "entre ríos", "entre rios", "neuquén", "neuquen", "salta"]:
  1117→        if prov_key in combined_lower:
  1118→            provincia = prov_key.title()
  1119→            break
  1120→
  1121→    # Category
  1122→    if "transfer" in combined_lower or "08" in combined_lower:
  1123→        categoria = "transfer"
  1124→    elif "multa" in combined_lower or "fotomulta" in combined_lower:
  1125→        categoria = "fine"
  1126→    elif "libre deuda" in combined_lower or "patente" in combined_lower:
  1127→        categoria = "vehicle_issue"
  1128→    else:
  1129→        categoria = "vehicle_issue"
  1130→
  1131→    # Verification level
  1132→    verification = "raw"
  1133→    if contact.get("contactability_score", 0) > 0 or provincia:
  1134→        verification = "enriched"
  1135→    if contact.get("contactability_score", 0) > 40 and provincia:
  1136→        verification = "validated"
  1137→
  1138→    # v7: timing — age_hours y stale detection
  1139→    dt_post = _parse_date(date)
  1140→    age_hours = None
  1141→    is_stale = False
  1142→    if dt_post:
  1143→        now_dt = datetime.now(timezone.utc)
  1144→        if dt_post.tzinfo is None:
  1145→            dt_post = dt_post.replace(tzinfo=timezone.utc)
  1146→        age_hours = round((now_dt - dt_post).total_seconds() / 3600, 1)
  1147→        if age_hours > STALE_THRESHOLD_HOURS:
  1148→            is_stale = True
  1149→
  1150→    lead_id = hashlib.sha256(f"{url}|{username}|{snippet[:100]}".encode()).hexdigest()[:16]
  1151→
  1152→    # Intent classification (v5.1)
  1153→    intent_cluster = classify_intent(combined)
  1154→
  1155→    lead = {
  1156→        "id": lead_id,
  1157→        "lead_origin": "real",
  1158→        "intent_cluster": intent_cluster,
  1159→        "entity_ref": entity.entity_id,
  1160→        "content": {"raw_text": snippet[:300], "normalized_text": combined_lower[:300]},
  1161→        "source": {"type": source_type, "provider": source, "url": url, "timestamp": datetime.now(timezone.utc).isoformat(), "query": query},
  1162→        "verification": {"level": verification, "confidence": 0.7 if verification == "validated" else (0.5 if verification == "enriched" else 0.3)},
  1163→        "timing": {"detected_at": datetime.now(timezone.utc).isoformat(), "age_hours": age_hours, "is_stale": is_stale},
  1164→        "contact_signals": {
  1165→            "whatsapp_detected": contact.get("whatsapp_detected", False),
  1166→            "whatsapp_validated": contact.get("whatsapp_validated", False),
  1167→            "phone_detected": contact.get("phone_detected", False),
  1168→            "contactability_score": contact.get("contactability_score", 0),
  1169→        },
  1170→        "persona": username or "(anónimo)",
  1171→        "provincia": provincia,
  1172→        "pais": "Argentina" if has_arg else "Unknown",
  1173→        "categoria": categoria,
  1174→        "platform": platform,
  1175→        "fecha_visible": date or datetime.now(timezone.utc).isoformat(),  # SPRINT 1: nunca vacío
  1176→        "contact": contact,
  1177→        "scoring": {},  # filled by Layer 4
  1178→        "provenance": {
  1179→            "raw_query": query,
  1180→            "provider_chain": [source],
  1181→            "transformation_steps": ["extraction", "normalization", "entity_linking", "scoring"],
  1182→        },
  1183→        "links": {"entity_id": entity.entity_id, "graph_neighbors": []},
  1184→        "actions": [],  # filled by Layer 5
  1185→    }
  1186→
  1187→    audit_log("entity_normalization", "lead_created", {"lead_id": lead_id, "entity_id": entity.entity_id, "intent_cluster": intent_cluster})
  1188→    return {"lead": lead, "entity": entity}
