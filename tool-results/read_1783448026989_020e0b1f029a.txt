   541→function extractPhone(text) {
   542→  const m = text.match(/(?:\\+54|0054)?[\\s\\-]?(?:9[\\s\\-]?)?(?:11|[2-9]\\d{2,3})[\\s\\-]?\\d{4}[\\s\\-]?\\d{4}/);
   543→  return m ? m[0].trim() : '';

   546→function normalizePhoneAR(raw) {
   547→  if (!raw) return { state: 'invalid_format', e164: '', display: '', waUrl: '' };
   548→  let digits = String(raw).replace(/[^0-9]/g, '');
   549→  if (!digits) return { state: 'invalid_format', e164: '', display: '', waUrl: '' };
   550→  if (digits.startsWith('54')) digits = digits.slice(2);
   551→  if (digits.startsWith('0')) digits = digits.slice(1);
   552→  // FIX GEMINI SABUESO v3: doble escape \\d dentro de template literal DASHBOARD_HTML.
   553→  // Sin esto, /^\d{3,4}15\d{6,7}$/ se convierte en /^d{3,4}15d{6,7}$/ que matchea
   554→  // el literal 'd' (no dígito) y rompe la normalización de teléfonos AR.
   555→  if (digits.length === 12 && /^\\d{3,4}15\\d{6,7}$/.test(digits)) {
   556→    digits = digits.replace(/15/, '');
   557→  }
   558→  if (digits.startsWith('9') && digits.length === 11) digits = digits.slice(1);
   559→  if (digits.length !== 10) return { state: 'invalid_format', e164: '', display: '', waUrl: '' };
   560→  const validAC = ['11','221','223','249','261','264','291','294','297','299','341','342','343','351','353','358','362','364','370','376','379','380','381','383','385','387','388'];
   561→  const ac3 = digits.slice(0, 3);
   562→  const ac2 = digits.slice(0, 2);
   563→  if (!validAC.includes(ac3) && !validAC.includes(ac2)) return { state: 'invalid_format', e164: '', display: '', waUrl: '' };
   564→  let mobile = digits;
   565→  if (digits.startsWith('11')) mobile = '9' + digits;
   566→  return {
   567→    state: 'normalized_contact',
   568→    e164: '+54' + mobile,
   569→    display: '+54 ' + mobile.slice(0, 2) + ' ' + mobile.slice(2, 6) + '-' + mobile.slice(6),
   570→    waUrl: 'https://wa.me/' + mobile,
   571→  };
   572→}

   589→function cleanSnippet(text) {
   590→  if (!text) return '';
   591→  let t = String(text);
   592→  // Quitar scripts de VentaFe (googletag, etc)
   593→  t = t.split('googletag')[0]; // Quitar googletag y todo lo que sigue
   594→  t = t.split('<script')[0]; // Quitar scripts
   595→  // Quitar comentarios HTML
   596→  t = t.replace(/<!--[\s\S]*?-->/g, '');
   597→  // Quitar tags HTML
   598→  t = t.replace(/<[^>]+>/g, ' ');
   599→  // Decodificar TODAS las entidades HTML
   600→  t = t.replace(/&amp;/g, '&').replace(/&lt;/g, '<').replace(/&gt;/g, '>').replace(/&quot;/g, '"').replace(/&#39;/g, "'");
   601→  t = t.replace(/&#x27;/g, "'").replace(/&#x2F;/g, '/').replace(/&#47;/g, '/');
   602→  t = t.replace(/&#8217;/g, "'").replace(/&#8230;/g, '...').replace(/&#8220;/g, '"').replace(/&#8221;/g, '"');
   603→  t = t.replace(/&#8211;/g, '-').replace(/&#8212;/g, '--').replace(/&#8216;/g, "'");
   604→  t = t.replace(/&#8242;/g, "'").replace(/&#8243;/g, '"').replace(/&nbsp;/g, ' ');
   605→  // Colapsar espacios
   606→  // FIX GEMINI SABUESO v3: doble escape \\s dentro de template literal DASHBOARD_HTML.
   607→  // Sin esto, el motor JS come la barra invertida y deja /s+/g que destruye todas las 's'
   608→  // del CRM ("Disputa" → "Di puta", "transferencia" → "tran ferencia", etc.)
   609→  t = t.replace(/\\s+/g, ' ').trim();
   610→  return t;
   611→}

   681→    if (aPin !== bPin) return aPin - bPin;
   682→    if (a.pinned && b.pinned) return (a.pinned_rank || 99) - (b.pinned_rank || 99);
   683→    // Resto: por heat_score o fecha
   684→    if (sort === 'heat') return (b._heat_score || 0) - (a._heat_score || 0);
   685→    if (sort === 'provincia') return (a.provincia || '').localeCompare(b.provincia || '');
   686→    return (b.fecha_iso || '').localeCompare(a.fecha_iso || '');
   687→  });
   688→
   689→  renderTable();
   690→  renderCounts();
   691→}
