   182→# HTTP helper con anti-blocking
   183→# ===========================================================================
   184→def _fetch_url(url: str, timeout: int = 15) -> Optional[str]:
   185→    """Fetch URL con random UA, retry y backoff."""
   186→    cache_key = hashlib.sha256(url.encode()).hexdigest()[:16]
   187→    now = time.time()
   188→
   189→    # Check cache
   190→    if cache_key in _cache:
   191→        ts, data = _cache[cache_key]
   192→        if now - ts < CACHE_TTL:
   193→            return data
   194→
   195→    for attempt in range(MAX_RETRIES):
   196→        try:
   197→            req = urllib.request.Request(url)
   198→            req.add_header("User-Agent", random.choice(USER_AGENTS))
   199→            req.add_header("Accept", "text/html,application/xhtml+xml,application/json,*/*")
   200→            req.add_header("Accept-Language", "es-AR,es;q=0.9,en;q=0.8")
   201→
   202→            with urllib.request.urlopen(req, timeout=timeout) as resp:
   203→                content = resp.read().decode("utf-8", errors="replace")
   204→                _cache[cache_key] = (now, content)
   205→                return content
   206→        except Exception as e:
   207→            if attempt < MAX_RETRIES - 1:
   208→                wait = (attempt + 1) * 2 + random.uniform(0, 1)
   209→                time.sleep(wait)
   210→            else:
   211→                return None
   212→    return None

   215→# ===========================================================================
   216→# Provider 1: DuckDuckGo HTML
   217→# ===========================================================================
   218→class _DDGResultParser(HTMLParser):
   219→    """Parser minimalista para resultados de DuckDuckGo HTML."""
   220→
   221→    def __init__(self):
   222→        super().__init__()
   223→        self.results: List[Dict[str, str]] = []
   224→        self._current: Dict[str, str] = {}
   225→        self._in_result = False
   226→        self._in_title = False
   227→        self._in_snippet = False
   228→        self._capture_tag = None
   229→
   230→    def handle_starttag(self, tag, attrs):
   231→        attrs_dict = dict(attrs)
   232→        # DuckDuckGo lite usa <a class="result-link">
   233→        # DuckDuckGo HTML usa <div class="result">
   234→        cls = attrs_dict.get("class", "")
   235→
   236→        if "result-link" in cls or ("result__a" in cls):
   237→            self._in_title = True
   238→            self._current = {"title": "", "url": "", "snippet": ""}
   239→            href = attrs_dict.get("href", "")
   240→            # DDG a veces wrappea URLs
   241→            if href.startswith("//duckduckgo.com/l/?uddg="):
   242→                href = urllib.parse.unquote(href.split("uddg=")[1].split("&")[0])
   243→            self._current["url"] = href
   244→
   245→        if "result-snippet" in cls or "result__snippet" in cls:
   246→            self._in_snippet = True
   247→            if not self._current:
   248→                self._current = {"title": "", "url": "", "snippet": ""}
   249→
   250→    def handle_endtag(self, tag):
   251→        if self._in_title and tag == "a":
   252→            self._in_title = False
   253→        if self._in_snippet and tag in ("a", "td", "div"):
   254→            self._in_snippet = False
   255→            if self._current and self._current.get("url"):
   256→                self.results.append(self._current)
   257→                self._current = {}
   258→
   259→    def handle_data(self, data):
   260→        if self._in_title:
   261→            self._current["title"] += data.strip()
   262→        if self._in_snippet:
   263→            self._current["snippet"] += data.strip() + " "

   414→# ===========================================================================
   415→# Search Manager — unifica todos los providers
   416→# ===========================================================================
   417→def search(query: str, num: int = 10) -> List[Dict[str, Any]]:
   418→    """
   419→    Busca usando múltiples providers en orden de fallback.
   420→
   421→    Orden:
   422→      1. DuckDuckGo (cobertura amplia)
   423→      2. Reddit (foros, alta calidad de leads)
   424→      3. RSS (backup)
   425→
   426→    Devuelve resultados normalizados con campos:
   427→      title, url, snippet, source, date, username?
   428→    """
   429→    all_results = []
   430→
   431→    # Provider 1: DuckDuckGo
   432→    try:
   433→        ddg = search_duckduckgo(query, num=num)
   434→        all_results.extend(ddg)
   435→        time.sleep(RATE_LIMIT_SECONDS)
   436→    except Exception:
   437→        pass
   438→
   439→    # Provider 2: Reddit (sólo si la query no es muy larga)
   440→    if len(query) < 200:
   441→        try:
   442→            reddit = search_reddit(query, num=num)
   443→            all_results.extend(reddit)
   444→            time.sleep(RATE_LIMIT_SECONDS)
   445→        except Exception:
   446→            pass
   447→
   448→    # Provider 3: RSS (sólo si los anteriores no dieron suficiente)
   449→    if len(all_results) < num // 2:
   450→        try:
   451→            rss = search_rss(query, num=num)
   452→            all_results.extend(rss)
   453→        except Exception:
   454→            pass
   455→
   456→    # Dedup por URL
   457→    seen_urls = set()
   458→    unique = []
   459→    for r in all_results:
   460→        url = r.get("url", "")
   461→        if url and url not in seen_urls:
   462→            seen_urls.add(url)
   463→            unique.append(r)
   464→
   465→    return unique[:num]

   655→def classify_intent(text: str) -> str:
   656→    """Clasifica el texto en un intent cluster IC-XXX."""
   657→    text_lower = text.lower()
   658→    best_cluster = None
   659→    best_score = 0
   660→    for cluster_id, cluster in INTENT_TAXONOMY.items():
   661→        score = sum(1 for kw in cluster["keywords"] if kw in text_lower)
   662→        if score > best_score:
   663→            best_score = score
   664→            best_cluster = cluster_id
   665→    return best_cluster or "IC-000"

   741→# Layer 4: Scoring engine (5 signals separados 0-100)
   742→# ===========================================================================
   743→def score_lead(text: str, date: str, has_contact: bool) -> Dict[str, Any]:
   744→    """Scoring con 5 signals independientes, cada uno 0-100."""
   745→    text_lower = text.lower()
   746→
   747→    # Signal 1: intent (lenguaje de intención explícita)
   748→    intent = 20
   749→    if any(kw in text_lower for kw in ["no puedo", "necesito", "quiero", "ayuda", "alguien sabe"]):
   750→        intent = 80
   751→    elif any(kw in text_lower for kw in ["consulta", "duda", "pregunta"]):
   752→        intent = 60
   753→
   754→    # Signal 2: urgency
   755→    urgency = 20
   756→    if any(kw in text_lower for kw in ["urgente", "hoy", "ya", "inmediato"]):
   757→        urgency = 90
   758→    elif any(kw in text_lower for kw in ["mañana", "vencimiento", "antes de"]):
   759→        urgency = 65
   760→
   761→    # Signal 3: financial_pressure
   762→    financial = 20
   763→    if any(kw in text_lower for kw in ["debo", "deuda", "multa", "$", "pesos", "plata"]):
   764→        financial = 75
   765→    if any(kw in text_lower for kw in ["bloqueada", "embargo", "inhibición"]):
   766→        financial = 85
   767→
   768→    # Signal 4: legal_risk
   769→    legal = 20
   770→    if any(kw in text_lower for kw in ["juzgado", "notificación", "citación", "inhibición"]):
   771→        legal = 85
   772→    elif any(kw in text_lower for kw in ["rechazaron", "bloqueada", "prohibido"]):
   773→        legal = 65
   774→
   775→    # Signal 5: recency
   776→    recency = 20
   777→    dt = _parse_date(date)
   778→    if dt:
   779→        now = datetime.now(timezone.utc)
   780→        if dt.tzinfo is None:
   781→            dt = dt.replace(tzinfo=timezone.utc)
   782→        diff = now - dt
   783→        if diff <= timedelta(days=3):
   784→            recency = 90
   785→        elif diff <= timedelta(days=7):
   786→            recency = 60
   787→
   788→    # Total: weighted sum clamped 0-100
   789→    # v6 weights exactos del spec
   790→    total = (
   791→        intent * 0.35 +
   792→        urgency * 0.20 +
   793→        financial * 0.20 +
   794→        legal * 0.15 +
   795→        recency * 0.10
   796→    )
   797→
   798→    # v6 modifiers
   799→    modifiers = {}
   800→    if has_contact:
   801→        modifiers["public_contact_visible"] = 15
   802→        total += 15
   803→    # TODO: multi_source_confirmation +25 when same entity appears in multiple sources
   804→    modifiers["multi_source_confirmation"] = 0
   805→
   806→    # v6 decay function: score * exp(-days_since_post / 7)
   807→    dt = _parse_date(date)
   808→    if dt:
   809→        now = datetime.now(timezone.utc)
   810→        if dt.tzinfo is None:
   811→            dt = dt.replace(tzinfo=timezone.utc)
   812→        days = (now - dt).total_seconds() / 86400
   813→        import math
   814→        decay = math.exp(-days / 7)
   815→        total = total * decay
   816→        modifiers["decay_factor"] = round(decay, 3)
   817→
   818→    total = max(0, min(100, int(round(total))))
   819→
   820→    # Penalties
   821→    penalties = {"duplication_risk": 0, "noise_probability": 0}
   822→    if not has_contact:
   823→        penalties["noise_probability"] = 30
   824→
   825→    # v7: multipliers
   826→    source_mult = 1.0  # default
   827→    # intent_multiplier from taxonomy
   828→    intent_mult = 1.0
   829→
   830→    return {
   831→        "total": total,
   832→        "base_score": 20,
   833→        "intent_multiplier": round(intent_mult, 2),
   834→        "source_multiplier": round(source_mult, 2),
   835→        "recency_multiplier": round(recency / 100, 2),
   836→        "final_score": total,
   837→        "signals": {
   838→            "intent": intent,
   839→            "urgency": urgency,
   840→            "financial_pressure": financial,
   841→            "legal_risk": legal,
   842→            "recency": recency,
   843→        },
   844→        "penalties": penalties,
   845→    }
   846→
   847→
   848→def _parse_date(date_str: str) -> Optional[datetime]:
   849→    if not date_str:
   850→        return None
   851→    for fmt in ["%Y-%m-%dT%H:%M:%S%z", "%Y-%m-%dT%H:%M:%SZ", "%Y-%m-%d", "%b %d, %Y"]:
   852→        try:
   853→            return datetime.strptime(date_str[:25], fmt)
   854→        except ValueError:
   855→            continue
   856→    return None

   967→def normalize_entity(result: Dict[str, Any], query: str) -> Optional[Dict[str, Any]]:
   968→    """Layer 2: normaliza un resultado en un lead + entity."""
   969→    name = result.get("name", "")
   970→    snippet = result.get("snippet", "")
   971→    url = result.get("url", "")
   972→    date = result.get("date", "")
   973→    source = result.get("source", "unknown")
   974→    username = result.get("username", "")
   975→    combined = f"{name}. {snippet}"
   976→    combined_lower = combined.lower()
   977→    host = get_host(url)
   978→
   979→    # Content filter
   980→    if not any(kw in combined_lower for kw in MUST_INCLUDE):
   981→        return None
   982→    for reject in REJECT_KW:
   983→        if reject in combined_lower:
   984→            return None
   985→    if any(d in host for d in INSTITUTIONAL_DOMAINS):
   986→        return None
   987→
   988→    # Country filter
   989→    for country, indicators in FOREIGN_INDICATORS.items():
   990→        for ind in indicators:
   991→            if ind in combined_lower:
   992→                return None
   993→
   994→    has_arg = any(s in combined_lower for s in ARG_SIGNALS)
   995→    if not has_arg:
   996→        has_arg = True  # assume Argentina if no foreign signal
   997→
   998→    # v6: source_type para policy engine
   999→    if "facebook" in host:
  1000→        source_type = "facebook_group"
